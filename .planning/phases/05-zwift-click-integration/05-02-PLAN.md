    3. `ClickShifter.on_notify(sender, data)` algorithm (plain `def`, NEVER `async def` — Pitfall 4):

    ```python
    def on_notify(self, sender, data) -> None:
        if not data or data[0] != CLICK_NOTIFY_MSG_TYPE:
            return  # idle, battery, or unknown message type
        # Walk the payload as (tag, value) pairs. Click frames are short (3-5 bytes).
        payload = bytes(data[1:])
        i = 0
        while i + 1 < len(payload):
            tag = payload[i]
            val = payload[i + 1]
            i += 2
            # Track per-button state so a release frame never shifts.
            prev = self._prev_button_state.get(tag, BUTTON_VALUE_RELEASED)
            self._prev_button_state[tag] = val
            if val != BUTTON_VALUE_PRESSED:
                continue            # release events: update state, don't shift
            if prev == BUTTON_VALUE_PRESSED:
                continue            # already pressed — no edge transition
            # Edge: released -> pressed. Apply debounce, then dispatch.
            now = self._clock()
            if (now - self._last_shift_t) < self._DEBOUNCE_S:
                continue
            self._last_shift_t = now
            if tag == PLUS_FIELD_TAG:
                self._gears.shift_up()
            elif tag == MINUS_FIELD_TAG:
                self._gears.shift_down()
    ```

    4. `ClickShifter.connect_and_listen(*, scanner, connect, stop_event, retry_backoff=2.0)`:

    ```python
    async def connect_and_listen(self, *, scanner=None, connect=None, stop_event, retry_backoff=DEFAULT_RETRY_BACKOFF_S) -> None:
        scanner = scanner or self._default_scanner
        connect = connect or self._default_connect
        while not stop_event.is_set():
            try:
                device = await scanner(timeout=DEFAULT_SCAN_TIMEOUT_S)
                if device is None:
                    _log.warning("Zwift Click not found; retrying in %.1fs", retry_backoff)
                    await asyncio.wait_for(stop_event.wait(), timeout=retry_backoff)
                    continue
                async with connect(device) as client:
                    await client.write_gatt_char(ZWIFT_SYNC_RX_CHAR_UUID, RIDE_ON, response=False)
                    await client.start_notify(ZWIFT_ASYNC_CHAR_UUID, self.on_notify)
                    _log.info("Zwift Click connected and notifying")
                    await stop_event.wait()
                    await client.stop_notify(ZWIFT_ASYNC_CHAR_UUID)
            except asyncio.TimeoutError:
                # stop_event.wait() above timed out — that means we should retry, not exit
                continue
            except BleakError as e:
                _log.warning("Click BLE error: %s — retrying in %.1fs", e, retry_backoff)
                try:
                    await asyncio.wait_for(stop_event.wait(), timeout=retry_backoff)
                except asyncio.TimeoutError:
                    pass
            except Exception as e:
                _log.warning("Click unexpected error: %s — retrying in %.1fs", e, retry_backoff)
                try:
                    await asyncio.wait_for(stop_event.wait(), timeout=retry_backoff)
                except asyncio.TimeoutError:
                    pass
    ```

    @staticmethod
    async def _default_scanner(*, timeout: float) -> "BLEDevice | None":
        from bleak.backends.scanner import AdvertisementData
        from bleak.backends.device import BLEDevice
        def _is_click(device: BLEDevice, adv: AdvertisementData) -> bool:
            mfr = adv.manufacturer_data.get(ZWIFT_MANUFACTURER_ID)
            if mfr and len(mfr) >= 1 and mfr[0] == CLICK_DEVICE_TYPE_BYTE:

            return ZWIFT_CUSTOM_SERVICE_UUID.lower() in [u.lower() for u in (adv.service_uuids or [])]
