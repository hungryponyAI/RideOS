    ```python
    """BLE reconnect lifecycle for the Wahoo KICKR Core.

    Phase 1: owned scan → connect → subscribe → wait-for-disconnect.
    Phase 2: adds FtmsController handshake + run_control_loop + shutdown-before-notify-stop
              inside the same async-with block (single BleakClient owner rule).
    """
    from __future__ import annotations

    import asyncio
    import logging
    from contextlib import AbstractAsyncContextManager
    from dataclasses import dataclass
    from typing import Awaitable, Callable, Optional

    from bleak import BleakClient, BleakError
    from bleak.backends.device import BLEDevice

    from engine.ble.client import (
        start_indoor_bike_notify,
        stop_indoor_bike_notify,
    )
    from engine.control.controller import FtmsController, run_control_loop
    from engine.control.state import RideState

    _log = logging.getLogger(__name__)

    FindDevice = Callable[[], Awaitable[Optional[BLEDevice]]]
    ConnectClient = Callable[
        [BLEDevice, Callable[[BleakClient], None]],
        AbstractAsyncContextManager[BleakClient],
    ]
    SleepFn = Callable[[float], Awaitable[None]]
    ControllerFactory = Callable[[BleakClient], FtmsController]
    OnClientReady = Callable[[BleakClient, FtmsController], None]


    @dataclass(frozen=True)
    class ReconnectConfig:
        initial_backoff: float = 1.0
        max_backoff: float = 60.0


    async def reconnect_loop(
        queue: "asyncio.Queue[Optional[bytes]]",
        find_device: FindDevice,
        connect_client: ConnectClient,
        config: ReconnectConfig = ReconnectConfig(),
        sleep: SleepFn = asyncio.sleep,
        stop_event: Optional[asyncio.Event] = None,
        *,
        ride_state: Optional[RideState] = None,
        controller_factory: ControllerFactory = FtmsController,
        on_client_ready: Optional[OnClientReady] = None,
    ) -> None:
        backoff = config.initial_backoff
        if stop_event is None:
            stop_event = asyncio.Event()  # safe local default

        while True:
            if stop_event.is_set():
                return

            try:
                device = await find_device()
            except (BleakError, OSError) as exc:
                _log.warning("Scan failed: %s (backoff=%.1fs)", exc, backoff)
                await sleep(backoff)
                backoff = min(backoff * 2, config.max_backoff)
                continue

            if device is None:
                if stop_event.is_set():
                    return
                _log.info("KICKR not found; retrying in %.1fs", backoff)
                await sleep(backoff)
                backoff = min(backoff * 2, config.max_backoff)
                continue

            _log.info("Connecting to %s", getattr(device, "name", device))

            disconnected = asyncio.Event()

            def _on_disconnect(_: BleakClient) -> None:
                disconnected.set()

            try:
                async with connect_client(device, _on_disconnect) as client:
                    await start_indoor_bike_notify(client, queue)
                    _log.info("Subscribed to FTMS Indoor Bike Data; awaiting data...")

                    controller: Optional[FtmsController] = None
                    control_task: Optional[asyncio.Task] = None

                    if ride_state is not None:
                        controller = controller_factory(client)
                        await controller.start()
                        if on_client_ready is not None:
                            on_client_ready(client, controller)
                        control_task = asyncio.create_task(
                            run_control_loop(controller, ride_state, stop_event),
                            name="ftms_control_loop",
                        )

                    try:
                        # Wait for EITHER disconnect OR shutdown.
                        stop_wait = asyncio.create_task(stop_event.wait())
                        disc_wait = asyncio.create_task(disconnected.wait())
                        try:
                            _, pending = await asyncio.wait(
                                {stop_wait, disc_wait},
                                return_when=asyncio.FIRST_COMPLETED,
                            )
                        finally:
                            for t in (stop_wait, disc_wait):
                                if not t.done():
                                    t.cancel()

                        if control_task is not None:
                            control_task.cancel()
                            try:
                                await control_task
                            except asyncio.CancelledError:
                                pass
                    finally:
                        # INFRA-02: Stop + Reset BEFORE stop_indoor_bike_notify
                        # and BEFORE the async-with exits (client still connected).
                        if controller is not None:
                            await controller.shutdown()

                    _log.info("Shutting down; stopping notifications cleanly")
                    try:
                        await stop_indoor_bike_notify(client)
                    except (BleakError, OSError):
                        pass
            except (BleakError, OSError) as exc:
                _log.warning(
                    "BLE error during connect/notify: %s (backoff=%.1fs)",
                    exc, backoff,
                )
                await sleep(backoff)
                backoff = min(backoff * 2, config.max_backoff)
                continue

            backoff = config.initial_backoff
            _log.info("Reconnect cycle complete; rescanning")
    ```

    ```python
        """Prove Phase 2 wiring: controller.shutdown() runs BEFORE stop_indoor_bike_notify."""
        from engine.gears.engine import GearEngine
        from engine.control.state import RideState
        # Note: we spy on stop_indoor_bike_notify via the client fake, and on
        # controller.shutdown via a spy subclass of FtmsController.

        call_order: list[str] = []

        class SpyController:
            def __init__(self, client):
                self._client = client
                self.controlled = False
            async def start(self):
                call_order.append("controller.start")
                self.controlled = True
            async def shutdown(self):
                call_order.append("controller.shutdown")
                self.controlled = False

        class FakeClient:
            async def start_notify(self, uuid, cb):
                call_order.append("start_notify")
            async def stop_notify(self, uuid):
                call_order.append("stop_notify")

        from contextlib import asynccontextmanager

        @asynccontextmanager
        async def fake_connect(device, on_disconnect):
            yield FakeClient()

        async def fake_find():
            from bleak.backends.device import BLEDevice
            # Minimal BLEDevice surrogate; we never touch its attrs
            class D: name = "KICKR CORE"
            return D()

        # stop after start_notify + controller.start have happened by waiting one loop turn.
        stop_event = asyncio.Event()

            await asyncio.sleep(0)
            await asyncio.sleep(0)
            stop_event.set()

        from engine.ble.reconnect import reconnect_loop
        from engine.ble.client import start_indoor_bike_notify, stop_indoor_bike_notify

        # Patch the module-level imports used inside reconnect_loop

        rc_stop = rc.stop_indoor_bike_notify

            call_order.append("start_indoor_bike_notify")
        async def spy_stop(client):
            call_order.append("stop_indoor_bike_notify")
        rc.start_indoor_bike_notify = spy_start
        rc.stop_indoor_bike_notify = spy_stop

                    stop_event=stop_event,

            await asyncio.wait_for(task, timeout=2.0)
        finally:
            rc.start_indoor_bike_notify = rc_start
            rc.stop_indoor_bike_notify = rc_stop

        # Key assertion: controller.shutdown runs BEFORE stop_indoor_bike_notify.
        shutdown_idx = call_order.index("controller.shutdown")
        notify_stop_idx = call_order.index("stop_indoor_bike_notify")

            f"controller.shutdown must run before stop_indoor_bike_notify "

        )
        # And controller.start ran before any grade writes.
        assert call_order.index("controller.start") < shutdown_idx
    ```

    ```python
    import logging
    import signal

    from engine.input.keyboard import KeyboardShifter

    DEFAULT_GRADE_PERCENT: float = 2.0

    async def _connect_client(device: BLEDevice, on_disconnect):

            device, disconnected_callback=on_disconnect
        ) as client:
            yield client

        speed = f"{reading.speed_kmh:5.1f} km/h" if reading.speed_kmh is not None else "  --  km/h"
        power = f"{reading.power_watts:4d} W" if reading.power_watts is not None else "  -- W"

            if reading.cadence_rpm is not None else "  --  rpm"
        )

    async def _gear_status_logger(state: RideState, stop_event: asyncio.Event) -> None:
        """Every 5s, log the current gear + effective grade — makes shifts visible without Phase 3 UI."""
        while not stop_event.is_set():
            gears = state.gear_engine
            eff = gears.effective_grade(state.real_grade_percent)
            _log.info(
                "RIDE | gear=%2d/%d  factor=%.3f  real=%.2f%%  effective=%.2f%%",

            level=logging.INFO,
            format="%(asctime)s %(levelname)s %(name)s: %(message)s",
        )

        queue: asyncio.Queue[Optional[bytes]] = asyncio.Queue()
        stop_event = asyncio.Event()

        loop = asyncio.get_running_loop()
        for sig in (signal.SIGINT, signal.SIGTERM):

                DEFAULT_GRADE_PERCENT,

                connect_client=_connect_client,

                stop_event=stop_event,
                ride_state=ride_state,
                on_client_ready=on_client_ready,
            ),
            name="reconnect_loop",
        )

        consumer_task = asyncio.create_task(
            telemetry_consumer(queue, _log_reading),
            name="telemetry_consumer",
        )

        status_task = asyncio.create_task(
            _gear_status_logger(ride_state, stop_event),
            name="gear_status_logger",
        )

        # Start the keyboard shifter AFTER the loop is running and BEFORE awaiting stop.
        try:
            shifter.start()
        except Exception as exc:  # noqa: BLE001
            _log.warning("KeyboardShifter start failed (non-interactive stdin?): %s", exc)

        try:
            await stop_event.wait()
            _log.info("Shutdown requested; stopping tasks")
            await queue.put(None)

            try:
                await asyncio.wait_for(
                    asyncio.gather(
                        reconnect_task, consumer_task, status_task,
                        return_exceptions=True,
                    ),
                    timeout=15.0,
                )
            except asyncio.TimeoutError:

                for t in (reconnect_task, consumer_task, status_task):
                    if not t.done():
                        t.cancel()
        finally:
            # ALWAYS restore terminal settings — even if the event loop crashed.
            try:
                shifter.stop()
            except Exception:
                _log.exception("KeyboardShifter.stop failed (terminal may be left in cbreak)")

        return 0


    if __name__ == "__main__":
        raise SystemExit(asyncio.run(main()))
    ```
